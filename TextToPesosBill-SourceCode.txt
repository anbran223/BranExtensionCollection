import android.app.Activity;
import android.content.Context;
import com.google.appinventor.components.annotations.*;
import com.google.appinventor.components.common.ComponentCategory;
import com.google.appinventor.components.runtime.AndroidNonvisibleComponent;
import com.google.appinventor.components.runtime.ComponentContainer;
import com.google.appinventor.components.runtime.EventDispatcher;

@DesignerComponent(
        version = 1,
        description = "",
        category = ComponentCategory.EXTENSION,
        nonVisible = true,
        iconName = "https://allmyassets.files.wordpress.com/2023/03/pixil-frame-0.png")

@SimpleObject(external = true)
//Libraries
@UsesLibraries(libraries = "")
//Permissions
@UsesPermissions(permissionNames = "")

public class TextToPesosBill extends AndroidNonvisibleComponent {

    //Activity and Context
    private Context context;
    private Activity activity;

    public TextToPesosBill(ComponentContainer container){
        super(container.$form());
        this.activity = container.$context();
        this.context = container.$context();
    }

    @SimpleFunction(description = "Converts a text string to Peso bill format")
    public String textToPesoBill(String text){
        StringBuilder result = new StringBuilder();
        int count = 0;
        for (int i = text.length() - 1; i >= 0; i--) {
            result.append(text.charAt(i));
            count++;
            if (count == 3 && i > 0) {
                result.append(',');
                count = 0;
            }
        }
        return "â‚± " + result.reverse().toString();
    }

    @SimpleFunction(description = "Converts a Peso bill format string to normal text")
    public String pesoBillToText(String pesoBill){
        String result = pesoBill.replace(",", "");
        return result.substring(2);
    }

    @SimpleEvent(description = "Test Event Generated by Niotron")
    public void testEvent(){
        EventDispatcher.dispatchEvent(this, "testEvent");
    }
}
